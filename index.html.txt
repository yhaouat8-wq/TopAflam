      /* ===== وظائف ===== */
      function renderGrid(filter = 'all', query = '') {
        moviesGrid.innerHTML = '';
        seriesGrid && (seriesGrid.innerHTML = '');
        const q = (query || '').trim().toLowerCase();
        const filtered = movies.filter(m => {
          if (filter === 'all') return q ? (m.title + m.desc + m.genre.join(' ')).toLowerCase().includes(q) : true;
          if (filter === 'series') return m.category === 'series';
          if (filter === 'kmovies') return m.category === 'kmovies';
          if (filter === 'kseries') return m.category === 'kseries';
          return m.category === filter;
        });
        // hero set to first item
        if (filtered.length) setHero(filtered[0]);
        else {
          heroPoster.src = '';
          heroTitle.textContent = 'ما كينش نتائج';
          heroMeta.textContent = '';
          heroDesc.textContent = '';
        }
        filtered.forEach(m => {
          const c = document.createElement('div');
          c.className = 'card';
          c.innerHTML = `
						<img loading="lazy" src="${m.img}" alt="${m.title}" />
						<div class="card-body">
							<div class="card-title">${m.title}</div>
							<div class="card-sub">${m.year} • ${m.genre[0]||''}</div>
						</div>`;
          c.onclick = () => openModal(m);
          if (m.category === 'series' || m.category === 'kseries') {
            seriesGrid && seriesGrid.appendChild(c);
          } else {
            moviesGrid.appendChild(c);
          }
        });
        if (filtered.length === 0) {
          moviesGrid.innerHTML = ' < div style = "color:var(--muted);padding:12px" > ما كاين حتى نتيجة. < /div>';
        }
      }

      function setHero(movie) {
        heroPoster.src = movie.img;
        heroTitle.textContent = movie.title;
        heroMeta.textContent = $ {
          movie.year
        }•
        $ {
          movie.genre.join('، ')
        }•
        $ {
          movie.duration
        };
        heroDesc.textContent = movie.desc;
        playHero.onclick = () => window.open(movie.watch, '_blank');
        moreHero.onclick = () => openModal(movie);
      }

      function openModal(movie) {
        modalPoster.src = movie.img;
        modalTitle.textContent = movie.title;
        modalMeta.textContent = $ {
          movie.year
        }•
        $ {
          movie.genre.join('، ')
        }•
        $ {
          movie.duration
        };
        modalTags.innerHTML = '';
        movie.genre.forEach(g => {
          const el = document.createElement('div');
          el.className = 'tag';
          el.textContent = g;
          modalTags.appendChild(el);
        });
        modalDesc.textContent = movie.desc;
        watchLink.href = movie.watch || '#';
        modal.classList.add('open');
        modal.setAttribute('aria-hidden', 'false');
        document.body.style.overflow = 'hidden';
      }

      function closeModalFn() {
        modal.classList.remove('open');
        modal.setAttribute('aria-hidden', 'true');
        document.body.style.overflow = '';
      }
      closeModal.onclick = closeModalFn;
      closeBtn.onclick = closeModalFn;
      modal.onclick = (e) => {
        if (e.target === modal) closeModalFn();
      }
      // nav links

          e.preventDefault();
          navLinks.forEach(x => x.classList.remove('active'));
          a.classList.add('active');
          const filter = a.getAttribute('data-filter');
          document.getElementById('seriesSection').style.display = (filter === 'series' || filter === 'kseries') ? 'block' : 'none';
